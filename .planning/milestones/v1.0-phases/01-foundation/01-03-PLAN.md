32-bit reconstruction (LOW-WORD-FIRST — PZEM-016 deviates from Modbus standard):
```
current = ((regs[2] as u32) << 16 | (regs[1] as u32)) as f64 / 1000.0
power   = ((regs[4] as u32) << 16 | (regs[3] as u32)) as f64 / 10.0
energy  = ((regs[6] as u32) << 16 | (regs[5] as u32)) as f64
```

  ```rust
  use anyhow::anyhow;

  #[derive(Debug, Clone)]
  pub struct PowerReading {
      pub device_name: String,
      pub voltage: f64,
      pub current: f64,
      pub power: f64,
      pub energy: f64,
      pub frequency: f64,
      pub power_factor: f64,
      pub timestamp_secs: i64,
  }

  pub fn decode_registers(regs: &[u16], device_name: &str) -> anyhow::Result<PowerReading> {
      if regs.len() < 10 {
          return Err(anyhow!("Expected 10 registers, got {}", regs.len()));
      }
      let voltage      = regs[0] as f64 / 10.0;
      let current      = ((regs[2] as u32) << 16 | regs[1] as u32) as f64 / 1000.0;
      let power        = ((regs[4] as u32) << 16 | regs[3] as u32) as f64 / 10.0;
      let energy       = ((regs[6] as u32) << 16 | regs[5] as u32) as f64;
      let frequency    = regs[7] as f64 / 10.0;
      let power_factor = regs[8] as f64 / 100.0;
      let timestamp_secs = std::time::SystemTime::now()
          .duration_since(std::time::UNIX_EPOCH)
          .unwrap_or_default()
          .as_secs() as i64;

      Ok(PowerReading {
          device_name: device_name.to_string(),
          voltage,
          current,
          power,
          energy,
          frequency,
          power_factor,
          timestamp_secs,
      })
  }
  ```

  Note on timestamp: `unwrap_or_default()` is safe here — `UNIX_EPOCH` is always before `now()`.
  The `SystemTime::now()` call in tests will be a live timestamp, but test assertions do not check
  timestamp value (only physical measurement fields are asserted).

  Do NOT add `chrono` — `std::time::SystemTime` is sufficient for epoch seconds.
