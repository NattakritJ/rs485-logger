tokio signal handling pattern (tokio "signal" feature already in Cargo.toml):
```rust
use tokio::signal;

// Create a one-shot shutdown signal that fires on SIGTERM or SIGINT
async fn shutdown_signal() {
    let ctrl_c = async {
        signal::ctrl_c().await.expect("failed to install Ctrl+C handler");
    };

    #[cfg(unix)]
    let terminate = async {
        signal::unix::signal(signal::unix::SignalKind::terminate())
            .expect("failed to install SIGTERM handler")
            .recv()
            .await;
    };

    #[cfg(not(unix))]
    let terminate = std::future::pending::<()>();

    tokio::select! {
        _ = ctrl_c => {},
        _ = terminate => {},
    }
}
```

Poll loop with graceful shutdown (tokio::select! on interval vs signal):
```rust
loop {
    tokio::select! {
        _ = ticker.tick() => {
            // poll all devices
        }
        _ = shutdown_signal() => {
            tracing::info!("Shutdown signal received, exiting after current cycle");
            break;
        }
    }
}
```

NOTE: `shutdown_signal()` is called fresh each iteration inside `tokio::select!`. For a cleaner approach, pin it outside the loop:
```rust
let shutdown = shutdown_signal();
tokio::pin!(shutdown);

loop {
    tokio::select! {
        _ = ticker.tick() => { /* poll */ }
        _ = &mut shutdown => {
            tracing::info!("Shutdown signal received");
            break;
        }
    }
}
```

tracing-subscriber init with EnvFilter + optional file appender:
```rust
use tracing_subscriber::{fmt, layer::SubscriberExt, util::SubscriberInitExt, EnvFilter};

// Basic stdout logging (OPS-02):
tracing_subscriber::fmt()
    .with_env_filter(EnvFilter::from_default_env())  // RUST_LOG env var, default "info"
    .init();

// With optional file appender (OPS-03):
let env_filter = EnvFilter::try_from_default_env()
    .unwrap_or_else(|_| EnvFilter::new("info"));

if let Some(log_path) = &cfg.log_file {
    let file_appender = tracing_appender::rolling::never(
        std::path::Path::new(log_path).parent().unwrap_or(std::path::Path::new(".")),
        std::path::Path::new(log_path).file_name().unwrap_or(std::os::unix::ffi::OsStrExt::as_bytes("rs485-logger.log") as _),
    );
    let (non_blocking, _guard) = tracing_appender::non_blocking(file_appender);
    tracing_subscriber::registry()
        .with(env_filter)
        .with(fmt::Layer::new().with_writer(non_blocking))
        .with(fmt::Layer::new().with_writer(std::io::stderr))
        .init();
} else {
    tracing_subscriber::fmt()
        .with_env_filter(env_filter)
        .with_writer(std::io::stderr)
        .init();
}
```

       let mut poller = ModbusPoller::new(&cfg.serial)?;
       let writer = InfluxWriter::new(&cfg.influxdb);

           std::time::Duration::from_secs(cfg.poll_interval_secs)

       let shutdown = shutdown_signal();
       tokio::pin!(shutdown);

           tokio::select! {

                       match poller.poll_device(device).await {
                           Ok(reading) => {
                               tracing::info!(device = %device.name, "Poll success");
