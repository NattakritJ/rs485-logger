Poll loop skeleton (implement exactly this pattern):
```rust
use tokio::time::{interval, Duration};

let cfg = load_config("config.toml")?;
let mut poller = ModbusPoller::new(&cfg.serial)?;
let writer = InfluxWriter::new(&cfg.influxdb);
let mut ticker = interval(Duration::from_secs(cfg.poll_interval_secs));

loop {
    ticker.tick().await;
    for device in &cfg.devices {
        match poller.poll_device(device).await {
            Ok(reading) => {
                if let Err(e) = writer.write(&reading).await {
                    tracing::warn!(device = %device.name, error = %e, "InfluxDB write failed");
                }
            }
            Err(e) => {
                tracing::warn!(device = %device.name, error = %e, "Device poll failed, skipping");
            }
        }
    }
}
```

    ```rust
    mod config;
    mod influx;
    mod poller;
    mod types;

    use config::load_config;
    use influx::InfluxWriter;
    use poller::ModbusPoller;

    #[tokio::main(flavor = "current_thread")]
    async fn main() -> anyhow::Result<()> {
        let cfg = load_config("config.toml")?;

        let mut poller = ModbusPoller::new(&cfg.serial)?;
        let writer = InfluxWriter::new(&cfg.influxdb);

        let mut ticker = tokio::time::interval(
            std::time::Duration::from_secs(cfg.poll_interval_secs)
        );

        loop {
            ticker.tick().await;
            for device in &cfg.devices {
                match poller.poll_device(device).await {
                    Ok(reading) => {
                        if let Err(e) = writer.write(&reading).await {
                            tracing::warn!(
                                device = %device.name,
                                error = %e,
                                "InfluxDB write failed"
                            );
                        }
                    }
                    Err(e) => {
                        tracing::warn!(
                            device = %device.name,
                            error = %e,
                            "Device poll failed, skipping"
                        );
                    }
                }
            }
        }
    }
    ```

    Key points:
    - `interval` ticks IMMEDIATELY on first call — the first poll happens at t=0, not t+interval. This is correct for a daemon that should poll on startup.
    - `tracing::warn!` with structured fields `device = %device.name` satisfies OPS-02 structured logging.
    - InfluxDB write errors use WARN (not ERROR) — a failed write is recoverable, the next interval will retry.
    - Device poll errors also WARN and continue (POLL-03).
    - No `#[allow(dead_code)]` needed — all types are now actively used.

    Run `cargo build` — must compile clean. The `#[allow(dead_code)]` attributes on config/influx/types modules will be resolved when the imports are live.
