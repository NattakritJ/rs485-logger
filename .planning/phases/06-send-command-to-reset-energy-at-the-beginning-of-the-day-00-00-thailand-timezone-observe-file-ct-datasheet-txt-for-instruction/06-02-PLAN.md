    ```rust
    /// Send FC 0x42 (Reset Energy) to a PZEM-016 device.
    ///
    /// Uses a 500 ms timeout (consistent with `poll_device`). On success,
    /// the device echoes the command back (`Response::Custom(0x42, _)`).
    /// On error, the device replies with `Response::Custom(0xC2, [abnormal_code])`.
    ///
    /// Returns `Ok(())` on success echo. Returns `Err` on timeout or transport
    /// failure. On a 0xC2 device error, logs WARN and returns `Ok(())` so the
    /// caller can continue to the next device (skip-and-log pattern, per D-02).
    pub async fn reset_energy(
        &mut self,
        device: &DeviceConfig,
    ) -> anyhow::Result<()> {
        self.ctx.set_slave(Slave(device.address));

        let result = tokio::time::timeout(
            Duration::from_millis(500),
            self.ctx.call(Request::Custom(0x42, Cow::Borrowed(&[]))),
        )
        .await
        .with_context(|| format!("Timeout resetting energy on device '{}'", device.name))?
        .with_context(|| format!("Modbus transport error resetting energy on device '{}'", device.name))?;

        match result {
            Response::Custom(0x42, _) => {
                // Success — device echoed the command back
                Ok(())
            }
            Response::Custom(0xC2, data) => {
                // Device returned an error reply — log WARN and continue (skip-and-log per D-02/D-12)
                let abnormal_code = data.first().copied().unwrap_or(0);
                tracing::warn!(
                    device = %device.name,
                    abnormal_code = abnormal_code,
                    "Energy reset failed — device returned 0xC2 error reply, skipping"
                );
                Ok(())
            }
            other => {
                Err(anyhow::anyhow!(
                    "Unexpected response to energy reset on device '{}': {:?}",
                    device.name,
                    other
                ))
            }
        }
    }
    ```

    **Step 4 — Rewrite the select! loop to add the third arm:**
    Replace the existing loop with:
    ```rust
    loop {
        tokio::select! {
            _ = ticker.tick() => {
                for device in &cfg.devices {
                    match poller.poll_device(device).await {
                        Ok(reading) => {
                            tracing::info!(device = %device.name, "Poll success");
                            if let Err(e) = writer.write(&reading).await {
                                tracing::warn!(device = %device.name, error = %e, "InfluxDB write failed");
                            }
                        }
                        Err(e) => {
                            tracing::warn!(device = %device.name, error = %e, "Device poll failed, skipping");
                        }
                    }
                }
            }
            // Daily energy reset arm — only active when reset_sleep is Some (per D-09)
            Some(sleep) = async {
                match reset_sleep.as_mut() {
                    Some(s) => { s.await; Some(()) }
                    None => None,
                }
            } => {
                let er = cfg.energy_reset.as_ref().unwrap(); // safe: reset_sleep Some implies reset_enabled
                tracing::info!("Daily energy reset starting"); // D-11
                for device in &cfg.devices {
                    match poller.reset_energy(device).await {
                        Ok(()) => {
                            tracing::info!(device = %device.name, "Energy reset OK"); // D-11
                        }
                        Err(e) => {
                            tracing::warn!(device = %device.name, error = %e, "Energy reset failed, skipping"); // D-12
                        }
                    }
                }
                // Recompute next reset (per D-08 — don't drift by adding 86400s)
                match next_reset_instant(Utc::now(), &er.timezone, &er.time) {
                    Ok(std_instant) => {
                        let next = tokio::time::Instant::from_std(std_instant);
                        let tz: chrono_tz::Tz = er.timezone.parse().unwrap_or(chrono_tz::Asia::Bangkok);
                        let delay = std_instant.duration_since(std::time::Instant::now()).unwrap_or_default();
                        let next_local = (Utc::now() + chrono::Duration::from_std(delay).unwrap_or_default())
                            .with_timezone(&tz);
                        tracing::info!(
                            next_reset = %next_local.format("%Y-%m-%dT%H:%M:%S%:z"),
                            "Next energy reset scheduled"
                        ); // D-13
                        reset_sleep = Some(Box::pin(tokio::time::sleep_until(next)));
                    }
                    Err(e) => {
                        tracing::warn!(error = %e, "Failed to recompute next reset — disabling energy reset");
                        reset_sleep = None;
                    }
                }
            }
            _ = &mut shutdown => {
                tracing::info!("Shutdown signal received, exiting cleanly");
                break;
            }
        }
    }
    ```

    **Note on tokio::select! with Option futures:** The pattern above uses an async block
    returning `Option<()>` for the reset arm. The `Some(sleep) =` pattern in select! will
    only match if the future returns `Some`, otherwise the arm is skipped — this is the
    standard tokio select! pattern for optional arms.

    If the async-block pattern causes compiler issues, use `futures::future::OptionFuture`
    or restructure using a `never()` future as the disabled case:
    ```rust

    // When reset_sleep is None, arm future is std::future::pending::<()>() 
    // which never resolves — effectively disabled.
    ```
